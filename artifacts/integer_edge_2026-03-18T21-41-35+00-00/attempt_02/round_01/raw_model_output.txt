```python
import argparse
import subprocess

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("binary", nargs="?", default=None)
    parser.add_argument("--binary", dest="binary_flag", default=None)
    args = parser.parse_args()
    binary = args.binary_flag or args.binary
    if not binary:
        raise SystemExit("missing binary path")

    proc = subprocess.Popen(
        [binary],
        stdin=subprocess.PIPE,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        text=False,
    )
    
    # The binary compares the input to 0xffffffc6 (-58 as a signed 32-bit integer).
    # Sending "-58" is robust because scanf parses it correctly into 0xffffffc6 
    # whether the format string is %d or %u.
    stdout, stderr = proc.communicate(input=b'-58\n', timeout=5)
    
    if stdout:
        print(stdout.decode(errors="replace"))
    if stderr:
        print(stderr.decode(errors="replace"))

if __name__ == "__main__":
    main()
```
